% Theorems
%
% redefines the beginning of a theorem to omit parentheses when
% an optional argument is given.
%
% Thus
%   \begin{theorem}[1] ... \end{theorem}
% results in
%   Theorem 1
% ( instead of Theorem (1) )
\setbeamertemplate{theorem begin}{%
  \begin{\inserttheoremblockenv}{%
      \inserttheoremname
      \ifx\inserttheoremaddition\@empty\else\ \inserttheoremaddition \fi%
    }%
}

% redefine footer
\setbeamertemplate{footline}{%
  \begin{beamercolorbox}[colsep=1.5pt]{upper separation line foot}
  \end{beamercolorbox}
 \begin{beamercolorbox}[ht=2.5ex,dp=1.125ex,
 leftskip=.3cm,rightskip=.3cm plus1fil]
{navigation symbols}%title in head/foot%}
 %%%

 %%%     \insertslidenavigationsymbol%
 %%%     \insertframenavigationsymbol%
%%%    }%
    \hfill
   {\insertframenumber / \inserttotalframenumber}
  \end{beamercolorbox}%
  \begin{beamercolorbox}[colsep=1.5pt]{lower separation line foot}
  \end{beamercolorbox}
}

  \end{beamercolorbox}
}
